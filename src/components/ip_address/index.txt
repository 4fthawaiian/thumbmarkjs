/**
 * This function takes a few hundred milliseconds to return.
 * Also, IP isn't necessarily a wise thing to include in a fingerprint, since it can change.
 * Using this depends on your use-case.
 * 
 * @returns the IP address
 */
export default async function fetchIpifyContent(): Promise<Object> {
    const url = 'http://api.ipify.org/';
  
    try {
      const response = await fetch(url);
      
      if (!response.ok) {
        throw new Error('Failed to fetch data');
      }
  
      const text = await response.text();
      return { 'ip_address': text };
    } catch (error: any) { // Explicitly defining the error as 'any' type
      return { 'ip_address': 'unknown'}
    }
  }
